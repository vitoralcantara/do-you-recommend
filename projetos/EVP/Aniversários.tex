\begin{zed}
   [NOME, ~ DATA] 
\end{zed}

\begin{zed} 
    RETORNO ::= ok | entrada\_existente | entrada\_nao\_existente
\end{zed}

\begin{schema}{Livro}
lNome : \power NOME \\
aniversario : NOME \pfun DATA
\where
	lNome = \dom aniversario
\end{schema}

 


\textbf{Tarefa 1:}
\\
\begin{schema}{EncontrarAnvisarioRobusto}
\Xi Livro \\
	nome?: NOME \\
	data!: DATA \\
	resultado!: RETORNO
\where
	(nome? \in lNome \land data! = aniversario(nome?) \land resultado! = ok)
	\lor \\
	(nome? \notin lNome \land resultado! = entrada\_nao\_existente \land data! = aniversario(nome?))
\end{schema}

\textbf{Tarefa 3:}

\begin{zed}
   [NOMES, ~ SENHA] 
\end{zed}

\begin{schema}{ContaUsuario}
	usuarios : \power NOMES \\
	senhaUsuario : NOMES \pfun SENHA \\
\where
	usuarios = \dom senhaUsuario	
\end{schema}

\begin{schema}{AdicionarUsuario}
\Delta ContaUsuario \\
	nome?: \power NOMES \\
	resultado!: RETORNO
\where
	((usuarios = usuarios \setminus nome?) \land usuarios' = usuarios \cup nome? \land  resultado! = ok)
	\lor \\
	(nome? \subset usuarios \land resultado! = entrada\_existente ) 
\end{schema}

\begin{schema}{DeletarUsuario}
\Delta ContaUsuario \\
	nome?: \power NOMES \\
\where
	usuarios' = usuarios \setminus nome? 
\end{schema}

\begin{schema}{ListaSenhaDeUsuario}
\Xi ContaUsuario \\
	nome?: NOMES \\
	senha!: SENHA \\
	resultado!: RETORNO
\where
	(nome? \in usuarios \land senha! = senhaUsuario(nome?) \land resultado! = ok)
	\lor \\
	(nome? \notin usuarios \land resultado! = entrada\_nao\_existente )	
\end{schema}
